#!/bin/bash
mem_size=$(free | awk '/Mem/{print $4}')

disk_size=$(df / | awk '/\//{print $4}')

i=1
while  [ $i -eq 1 ] 
do
        if [ $disk_size -le 51200 -o $mem_size -le 102400000 ]
        then
                echo "warning !"
        fi
let i++
done

                echo "192.168.39.$i is up"
        else
                echo "192.168.39.$i is down"
        fi
done
